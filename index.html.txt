    <script>
        // --- Referências aos Elementos ---
        const mainContainer = document.getElementById('main-container');
        const botaoSimPrincipal = document.getElementById('botao-sim-principal');
        const botaoNaoPrincipal = document.getElementById('botao-nao-principal');
        const rejectionModal = document.getElementById('rejectionModal');
        const modalContent = document.getElementById('modal-content');
        const modalBotaoSim = document.getElementById('modal-botao-sim');
        const modalBotaoNao = document.getElementById('modal-botao-nao');
        const mensagemFinal = document.getElementById('mensagem-final');

        // --- Funções Auxiliares ---

        function fecharJanelaPrincipal() {
            mainContainer.style.display = 'none';
        }

        function mostrarMensagemFinal() {
            fecharJanelaPrincipal();
            rejectionModal.style.display = 'none'; // Garante que o modal esteja fechado
            mensagemFinal.style.display = 'block';
        }

        function moverBotaoModal(event) {
            // Pega as dimensões do conteúdo do modal (onde o botão pode se mover)
            const larguraModal = modalContent.offsetWidth;
            const alturaModal = modalContent.offsetHeight;

            // Dimensões do botão 'Sim' do modal
            const larguraBotao = modalBotaoSim.offsetWidth;
            const alturaBotao = modalBotaoSim.offsetHeight;

            const margem = 10;
            
            // Calcula novas posições aleatórias dentro dos limites do modal
            const novaX = Math.random() * (larguraModal - larguraBotao - 2 * margem) + margem;
            // Define o limite superior para não cobrir o título e o limite inferior para não cobrir o botão 'Não'
            const limiteSuperiorY = 60; 
            const limiteInferiorY = alturaModal - alturaBotao - 70; 
            const novaY = Math.random() * (limiteInferiorY - limiteSuperiorY) + limiteSuperiorY;

            // Aplica as novas posições
            modalBotaoSim.style.left = novaX + 'px';
            modalBotaoSim.style.top = novaY + 'px';
        }


        // --- Event Listeners ---

        // 1. Ação "Sim" Principal (Resposta correta de primeira)
        botaoSimPrincipal.addEventListener('click', mostrarMensagemFinal);

        // 2. Ação "Não" Principal (Abre o Pop-up de certeza)
        botaoNaoPrincipal.addEventListener('click', () => {
            rejectionModal.style.display = 'flex';
            // Garante que o botão 'Sim' do modal comece em uma posição neutra
            modalBotaoSim.style.left = '50%'; 
            modalBotaoSim.style.top = '70px'; 
        });

        // 3. Ação 'Sim' do Pop-up (Iniciar a Fuga)
        // O botão 'Sim' (para confirmar a recusa) FOGE ao passar o mouse/tocar
        modalBotaoSim.addEventListener('mouseover', moverBotaoModal);
        modalBotaoSim.addEventListener('touchstart', (e) => {
            e.preventDefault();
            moverBotaoModal(e);
        });

        // 4. Ação 'Não' do Pop-up (Desistir da Recusa)
        // Clicar em "Não" no pop-up fecha o pop-up e retorna à tela principal.
        modalBotaoNao.addEventListener('click', () => {
            rejectionModal.style.display = 'none';
        });

    </script>
